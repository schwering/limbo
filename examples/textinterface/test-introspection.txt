Sort OBJ;
Var x -> OBJ;
Name n -> OBJ;

Query: n = n;
Query: Ex x x = x;
Query: Fa x x = x;
Assert: Ex x K<0> x = x;
Assert: Ex x K<0> x = n;
Assert: Fa x K<0> x = x;
Refute: Fa x K<0> x = n;
